145 void Scene::Trigger()
146 {
147 int manifold_count = dynamicsWorld->getDispatcher()->getNumManifolds();
148
149 for (int i = 0; i < manifold_count; i++)
150 {
151 btPersistentManifold* manifold = dynamicsWorld->getDispatcher()->getManifoldByIndexInternal(i);
152
153 Entity* object_a = static_cast<Entity*>(manifold->getBody0()->getUserPointer());
154 Entity* object_b = (Entity*)(manifold->getBody1()->getUserPointer());
155
156 object_a->IsCollision(object_b);
157 object_b->IsCollision(object_a);
158 }
